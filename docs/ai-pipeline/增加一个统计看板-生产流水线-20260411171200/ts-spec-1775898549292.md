// interceptors/permission.interceptor.ts
function intercept(request) {
  const user = extractUserFromToken(request);
  
  if (!user) return 401;
  
  if (user.role === 'department_manager') {
    request.departmentIds = user.managedDepartments;
  } else if (user.role === 'admin') {
    request.departmentIds = null; // no restriction
  } else {
    return 403;
  }
  
  return next();
}

### 4.2 KPI计算服务
```typescript
// services/kpi-calculator.service.ts
class KpiCalculatorService {
  calculateKpi(query: DashboardQueryDto, departmentIds: string[] | null): KpiResponseDto {
    // 1. 获取基础数据
    const stats = this.worklogRepository.getStats({
      startDate: query.startDate,
      endDate: query.endDate,
      departmentIds
    });
    
    // 2. 计算提交率
    let submissionRate = null;
    if (stats.requiredTotal > 0) {
      submissionRate = (stats.submittedTotal / stats.requiredTotal) * 100;
      submissionRate = Math.round(submissionRate * 10) / 10; // 保留一位小数
    }
    
    // 3. 计算及时率
    let onTimeRate = null;
    if (stats.submittedTotal > 0) {
      onTimeRate = (stats.onTimeTotal / stats.submittedTotal) * 100;
      onTimeRate = Math.round(onTimeRate * 10) / 10;
    }
    
    // 4. 计算平均时长
    let avgDuration = null;
    let formattedDuration = undefined;
    if (stats.submittedTotal > 0) {
      avgDuration = Math.round(stats.totalDuration / stats.submittedTotal);
      
      if (avgDuration >= 60) {
        const hours = Math.floor(avgDuration / 60);
        const minutes = avgDuration % 60;
        formattedDuration = `${hours}小时${minutes}分钟`;
      }
    }
    
    return {
      submissionRate,
      onTimeRate,
      avgDuration,
      formattedDuration
    };
  }
}
```

// services/trend-aggregator.service.ts
class TrendAggregatorService {
  aggregateTrend(query: DashboardQueryDto, departmentIds: string[] | null): TrendPointDto[] {
    const rawData = this.worklogRepository.getDailyStats({
      startDate: query.startDate,
      endDate: query.endDate,
      departmentIds
    });
    
    if (query.granularity === 'day') {
      return rawData.map(day => ({
        date: day.date,
        submissionRate: day.requiredCount > 0 
          ? Math.round((day.submittedCount / day.requiredCount) * 1000) / 10
          : 0,
        onTimeRate: day.submittedCount > 0
          ? Math.round((day.onTimeCount / day.submittedCount) * 1000) / 10
          : 0
      }));
    }
    
    // 周/月聚合逻辑类似
    return this.aggregateByGranularity(rawData, query.granularity);
  }
}

### 4.4 对比数据获取
```typescript
// services/comparison.service.ts
class ComparisonService {
  getComparisonData(query: DashboardQueryDto, departmentIds: string[] | null): ComparisonItemDto[] {
    if (query.dimension === 'dept') {
      return this.getDepartmentComparison(query, departmentIds);
    } else {
      return this.getEmployeeComparison(query, departmentIds);
    }
  }
  
  private getEmployeeComparison(query: DashboardQueryDto, departmentIds: string[] | null): ComparisonItemDto[] {
    const employeeStats = this.worklogRepository.getEmployeeStats({
      startDate: query.startDate,
      endDate: query.endDate,
      departmentIds
    });
    
    // 计算提交率并排序
    const items = employeeStats.map(stat => {
      const submissionRate = stat.requiredCount > 0
        ? Math.round((stat.submittedCount / stat.requiredCount) * 1000) / 10
        : 0;
      
      const onTimeRate = stat.submittedCount > 0
        ? Math.round((stat.onTimeCount / stat.submittedCount) * 1000) / 10
        : 0;
      
      return {
        id: stat.employeeId,
        name: stat.employeeName,
        submissionRate,
        onTimeRate
      };
    });
    
    // 按提交率升序，取Top 20
    return items
      .sort((a, b) => (a.submissionRate || 0) - (b.submissionRate || 0))
      .slice(0, 20);
  }
}
```

## 5. 异常处理
- 错误规则：
  1. 数据库查询失败 → 返回500，日志记录
  2. 参数验证失败 → 返回400，包含具体错误字段
  3. 权限不足 → 返回403，不暴露具体原因
  4. 数据为空 → 返回200，data为null或空数组
  5. 服务依赖失败 → 返回503，启用降级策略
