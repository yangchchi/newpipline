## 4. 核心流程
### 4.1 提交日报流程
```typescript
async function submitDailyReport(userId: string, dto: SubmitDailyReportDTO): Promise<DailyReport> {
    // 1. 验证输入
    validateSubmitDTO(dto);
    
    // 2. 检查是否已提交
    const existing = await dailyReportRepo.findOne({
        where: {
            userId,
            reportDate: new Date(dto.reportDate),
            status: "SUBMITTED"
        }
    });
    if (existing) throw new ConflictError("当日报告已提交");
    
    // 3. 计算总工时
    const totalHours = dto.tasks.reduce((sum, task) => sum + task.hours, 0);
    
    // 4. 创建或更新日报
    let report = await dailyReportRepo.findOne({
        where: {
            userId,
            reportDate: new Date(dto.reportDate),
            status: "DRAFT"
        },
        relations: ["tasks"]
    });
    
    if (!report) {
        report = dailyReportRepo.create({
            userId,
            reportDate: new Date(dto.reportDate),
            status: "DRAFT",
            totalHours: 0
        });
        await dailyReportRepo.save(report);
    }
    
    // 5. 更新任务
    await taskRepo.delete({ dailyReportId: report.id });
    
    const tasks = dto.tasks.map(task => taskRepo.create({
        ...task,
        dailyReportId: report.id
    }));
    await taskRepo.save(tasks);
    
    // 6. 更新日报状态
    report.status = "SUBMITTED";
    report.totalHours = totalHours;
    report.submittedAt = new Date();
    await dailyReportRepo.save(report);
    
    // 7. 触发PM系统同步（异步）
    triggerPMSync(report.id);
    
    return report;
}

function validateSubmitDTO(dto: SubmitDailyReportDTO): void {
    // 任务列表非空
    if (!dto.tasks || dto.tasks.length === 0) {
        throw new ValidationError("请至少添加一条任务记录");
    }
    
    // 每项任务都有项目
    const invalidTasks = dto.tasks.filter(t => !t.projectId);
    if (invalidTasks.length > 0) {
        throw new ValidationError("请为每项任务关联所属项目");
    }
    
    // 计算总工时
    const totalHours = dto.tasks.reduce((sum, task) => sum + task.hours, 0);
    
    // 总工时范围
    if (totalHours <= 0) throw new ValidationError("总工时需大于0");
    if (totalHours > 24) throw new ValidationError("总工时不能超过24小时");
}
```

async function getProjects(search?: string, activeOnly = true): Promise<ProjectDTO[]> {
    const query = projectRepo.createQueryBuilder("project");
    
    if (activeOnly) {
        query.where("project.isActive = :isActive", { isActive: true });
    }
    
    if (search) {
        query.andWhere("project.name ILIKE :search", { search: `%${search}%` });
    }
    
    query.orderBy("project.name", "ASC");
    
    const projects = await query.getMany();
    return projects.map(p => ({ id: p.id, name: p.name }));
}

## 5. 异常处理
### 错误规则
```typescript
class ValidationError extends Error {
    constructor(message: string) {
        super(message);
        this.name = "ValidationError";
    }
}

class ConflictError extends Error {
    constructor(message: string) {
        super(message);
        this.name = "ConflictError";
    }
}

// Express错误处理中间件
app.use((err: Error, req: Request, res: Response, next: NextFunction) => {
    if (err instanceof ValidationError) {
        return res.status(400).json({ success: false, error: err.message });
    }
    if (err instanceof ConflictError) {
        return res.status(409).json({ success: false, error: err.message });
    }
    console.error(err);
    res.status(500).json({ success: false, error: "服务器内部错误" });
});
```
